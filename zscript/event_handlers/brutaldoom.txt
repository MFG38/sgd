class SGDBrutalDoomHandler : StaticEventHandler
{
	bool detected;

	override void OnRegister()
	{
		for (int i = 0; i < AllActorClasses.size(); i++)
		{
			if ((AllActorClasses[i].GetClassName() != "BrutalWeapon") && (AllActorClasses[i].GetClassName() != "BrutalDoomer"))
			{
				continue;
			}
			detected = true;
			break;
		}
	}
}
