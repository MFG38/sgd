class DootalHandler : StaticEventHandler
{
    bool BDDetected;

    override void OnRegister()
    {
        for(int i = 0; i < AllActorClasses.size(); i++)
        {
            if( AllActorClasses[i].GetClassName() == "BDoomer" ||
                AllActorClasses[i].GetClassName() == "BrutalDoomer"
            )
            {
                BDDetected = true;
            }
        }
        if(BDDetected) { break; }
    }
    if(!BDDetected) { return; }

    override void WorldTick()
    {
        if(!BDDetected) { return; }

        S_StartSound("skeleton/sight",999,CHANF_UI|CHANF_OVERLAP,1.0);
        Console.Log(StringTable.Localize("$SGD_WRN_BRUTAL"));
    }
}
