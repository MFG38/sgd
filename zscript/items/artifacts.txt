class SGDMegasphereAlt : SGDMegasphere
{
	Default { Inventory.PickupMessage "$SGD_ART_MEGASPHEREALT"; }
	States
	{
	Pickup:
		TNT1 A 0 A_GiveInventory("AltMegasphereHealth",1);
		TNT1 A 0 A_GiveInventory("AltMegasphereArmor",1);
		Stop;
	}
}

class AltMegasphereHealth : Health
{
	Default { Inventory.Amount 100; Inventory.MaxAmount 200; }
}

class AltMegasphereArmor : BasicArmorPickup
{
	Default { Armor.SavePercent 50; Armor.SaveAmount 100; }
}
