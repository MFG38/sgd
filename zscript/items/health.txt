class SGDHealthBonus : HealthBonus replaces HealthBonus
{
	Default
	{
		Inventory.PickupMessage "$SGD_ART_HEALTHBONUS";
		Inventory.ForbiddenTo "DemonizedPlayer";
		-INVENTORY.ALWAYSPICKUP;
	}
}
