class SGDCacodemon : SGDMonster replaces Cacodemon
{
	Default
	{
		Health 400;
		Radius 31;
		Height 56;
		Mass 400;
		Speed 8;
		PainChance 128;
		+FLOAT;
		+NOGRAVITY;
		-FLOORCLIP;
		SeeSound "caco/sight";
		PainSound "caco/pain";
		DeathSound "caco/death";
		ActiveSound "caco/active";
		Obituary "$OB_CACO";
		HitObituary "$OB_CACOHIT";
		BloodColor "00 00 80";
	}
	States
	{
	Spawn:
		HEAD A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		HEAD A 10 A_Look;
		Loop;
	AltSpawn:
		HEAD A 3 A_Wander;
		HEAD A 0 A_Look;
		Loop;
	See:
		HEAD A 3 A_Chase;
		Loop;
	Missile:
		HEAD B 5 A_FaceTarget;
		HEAD C 5 A_FaceTarget;
		HEAD D 5 BRIGHT A_HeadAttack;
		Goto See;
	Pain:
		HEAD E 3;
		HEAD E 3 A_Pain;
		HEAD F 6;
		Goto See;
	Pain.HolyWater:
		HEAD E 3 A_SetTranslation("HolyWaterTrans");
		HEAD E 3 A_Pain;
		HEAD F 6;
		Goto See;
	Pain.Fire:
		HEAD E 3 A_SetTranslation("CharredTrans");
		HEAD E 3 A_Pain;
		HEAD F 6;
		Goto See;
	Death:
		HEAD G 8
		{
			A_GiveToTarget("CacodemonsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		HEAD H 8 A_Scream;
		HEAD I 8;
		HEAD J 8;
		HEAD K 8 A_NoBlocking;
		HEAD L -1 A_SetFloorClip;
		Stop;
	Death.HolyWater:
		HEAD G 8
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("CacodemonsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		HEAD H 8 A_Scream;
		HEAD I 8;
		HEAD J 8;
		HEAD K 8 A_NoBlocking;
		HEAD L -1 A_SetFloorClip;
		Stop;
	Death.Fire:
		HEAD G 8
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("CacodemonsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		HEAD H 8 A_Scream;
		HEAD I 8;
		HEAD J 8;
		HEAD K 8 A_NoBlocking;
		HEAD L -1 A_SetFloorClip;
		Stop;
	Death.Sploder:
		HEAD E 3 A_Pain;
		HEAD EEEEEEEEEE 5;
		HEAD G 8
		{
			A_GiveToTarget("CacodemonsKilled");
			A_GiveToTarget("FinisherToken",25);
			A_SpawnItemEx("Explosion",0,0,35);
			//if(FindInventory("ChainExploder")) { A_RadiusGive("ChainExploder",192,RGF_MONSTERS); }
		}
		HEAD H 8
		{
			A_Scream();
			A_Explode(666,192,XF_NOTMISSILE);
		}
		HEAD I 8;
		HEAD J 8;
		HEAD K 8 A_NoBlocking;
		HEAD L -1 A_SetFloorClip;
		Stop;
	Raise:
		HEAD L 8
		{
			A_SetTranslation("ResetTrans");
			A_UnSetFloorClip();
		}
		HEAD KJIHG 8;
		Goto See;
	}
}

class SGDCacodemonBall : CacodemonBall replaces CacodemonBall
{
	Default
	{
		Speed 16;
	}

	States
	{
	Spawn:
		BAL2 AAAABBBB 1 BRIGHT { if(CVar.FindCVar('sgd_spawntrails').GetBool() == 1) { A_SpawnItemEx("SGDProjectileTrail",-20,0,-vel.z); }}
		Loop;
	}
}
