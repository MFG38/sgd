class SGDCyberdemon : SGDMonster replaces Cyberdemon
{
	Default
	{
		Health 4000;
		Radius 40;
		Height 110;
		Mass 1000;
		Speed 16;
		PainChance 20;
		+BOSS;
		+MISSILEEVENMORE;
		+NORADIUSDMG;
		+DONTMORPH;
		+BOSSDEATH;
		SeeSound "cyber/sight";
		PainSound "cyber/pain";
		DeathSound "cyber/death";
		ActiveSound "cyber/active";
		Obituary "$OB_CYBORG";
		PainChance "Fire", 40;
		PainChance "HolyWater", 64;
		DamageFactor "Sawblade", 2.5;
	}
	States
	{
	Spawn:
		//CYBR A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		CYBR AB 10 A_Look;
		Loop;
	AltSpawn:
		CYBR A 3 A_Hoof;
		CYBR ABBCC 3 A_Wander;
		CYBR D 3 A_Metal;
		CYBR D 3 A_Wander;
		CYBR A 0 A_Look;
		Loop;
	See:
		CYBR A 3 A_Hoof;
		CYBR ABBCC 3 A_Chase;
		CYBR D 3 A_Metal;
		CYBR D 3 A_Chase;
		Loop;
	Missile:
		CYBR E 2 A_FaceTarget;
		CYBR F 10 A_CyberAttack;
		CYBR E 10 A_FaceTarget;
		CYBR F 10 A_CyberAttack;
		CYBR E 10 A_FaceTarget;
		CYBR F 10 A_CyberAttack;
		Goto See;
	Pain:
		CYBR G 10 A_Pain;
		Goto See;
	Pain.HolyWater:
		CYBR G 0 A_SetTranslation("HolyWaterTrans");
		CYBR G 10 A_Pain;
		Goto See;
	Pain.Fire:
		CYBR G 0 A_SetTranslation("CharredTrans");
		CYBR G 10 A_Pain;
		Goto See;
	Death:
		CYBR H 10
		{
			A_GiveToTarget("CyberdemonsKilled");
			A_GiveToTarget("FinisherToken",100);
		}
		CYBR I 10 A_Scream;
		CYBR JKL 10;
		CYBR M 10 A_NoBlocking;
		CYBR NO 10;
		CYBR P 30;
		CYBR P -1 A_BossDeath;
		Stop;
	Death.HolyWater:
		CYBR H 10
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("CyberdemonsKilled");
			A_GiveToTarget("FinisherToken",100);
		}
		CYBR I 10 A_Scream;
		CYBR JKL 10;
		CYBR M 10 A_NoBlocking;
		CYBR NO 10;
		CYBR P 30;
		CYBR P -1 A_BossDeath;
		Stop;
	Death.Fire:
		CYBR H 10
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("CyberdemonsKilled");
			A_GiveToTarget("FinisherToken",100);
		}
		CYBR I 10 A_Scream;
		CYBR JKL 10;
		CYBR M 10 A_NoBlocking;
		CYBR NO 10;
		CYBR P 30;
		CYBR P -1 A_BossDeath;
		Stop;
	Death.Sploder:
		CYBR G 2 A_Pain;
		CYBR GGGGGGGGGG 5;
		CYBR H 10
		{
			A_GiveToTarget("CyberdemonsKilled");
			A_GiveToTarget("FinisherToken",100);
			
			for(int a = 0; a < 4; a++)
			{
				A_SpawnItemEx("Explosion",random(-16,16),random(-16,16),random(52,80));
			}
		}
		CYBR I 10
		{
			A_Scream();
			A_Explode(666,192,XF_NOTMISSILE);
		}
		CYBR JKL 10;
		CYBR M 10 A_NoBlocking;
		CYBR NO 10;
		CYBR P 30;
		CYBR P -1 A_BossDeath;
		Stop;
	}
}

	}

	}
}
