class SGDRevenant: SGDMonster replaces Revenant
{
	Default
	{
		Health 300;
		Radius 20;
		Height 56;
		Mass 500;
		Speed 10;
		PainChance 100;
		Monster;
		MeleeThreshold 196;
		SeeSound "skeleton/sight";
		PainSound "skeleton/pain";
		DeathSound "skeleton/death";
		ActiveSound "skeleton/active";
		MeleeSound "skeleton/melee";
		HitObituary "$OB_UNDEADHIT";
		Obituary "$OB_UNDEAD";
	}
	States
	{
	Spawn:
		SKEL A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		SKEL AB 10 A_Look;
		Loop;
	AltSpawn:
		SKEL AABBCCDDEEFF 2 A_Wander;
		SKEL A 0 A_Look;
		Loop;
	See:
		SKEL AABBCCDDEEFF 2 A_Chase;
		Loop;
	Melee:
		SKEL G 0 A_FaceTarget;
		SKEL G 6 A_SkelWhoosh;
		SKEL H 6 A_FaceTarget;
		SKEL I 6 A_SkelFist;
		Goto See;
	Missile:
		SKEL J 0 A_JumpIf(GetCVar("sgd_revrocketbehavior") == 1,"AltMissile");
		SKEL J 0 BRIGHT A_FaceTarget;
		SKEL J 5 BRIGHT A_FaceTarget;
		SKEL K 10 A_SkelMissile;
		SKEL K 10 A_FaceTarget;
		Goto See;
	AltMissile:
		SKEL J 0 BRIGHT A_FaceTarget;
		SKEL J 5 BRIGHT A_FaceTarget;
		SKEL K 10 A_SkelAltMissile;
		SKEL K 10 A_FaceTarget;
		Goto See;
	Pain:
		SKEL L 5;
		SKEL L 5 A_Pain;
		Goto See;
	Pain.HolyWater:
		SKEL L 5 A_SetTranslation("HolyWaterTrans");
		SKEL L 5 A_Pain;
		Goto See;
	Pain.Fire:
		SKEL L 5 A_SetTranslation("CharredTrans");
		SKEL L 5 A_Pain;
		Goto See;
	Death:
		SKEL L 7
		{
			A_GiveToTarget("RevenantsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		SKEL M 7;
		SKEL N 7 A_Scream;
		SKEL O 7 A_NoBlocking;
		SKEL P 7;
		SKEL Q -1;
		Stop;
	Death.HolyWater:
		SKEL L 7
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("RevenantsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		SKEL M 7;
		SKEL N 7 A_Scream;
		SKEL O 7 A_NoBlocking;
		SKEL P 7;
		SKEL Q -1;
		Stop;
	Death.Fire:
		SKEL L 7
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("RevenantsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		SKEL M 7;
		SKEL N 7 A_Scream;
		SKEL O 7 A_NoBlocking;
		SKEL P 7;
		SKEL Q -1;
		Stop;
	Death.Sploder:
		SKEL L 2 A_Pain;
		SKEL LLLLLLLLLL 5;
		SKEL L 7
		{
			A_GiveToTarget("RevenantsKilled");
			A_GiveToTarget("FinisherToken",25);
			A_SpawnItemEx("Explosion",0,0,35);
			//if(FindInventory("ChainExploder")) { A_RadiusGive("ChainExploder",192,RGF_MONSTERS); }
		}
		SKEL M 7;
		SKEL N 7
		{
			A_Scream();
			A_Explode(666,192,XF_NOTMISSILE);
		}
		SKEL O 7 A_NoBlocking;
		SKEL P 7;
		SKEL Q -1;
		Stop;
	Raise:
		SKEL Q 5 A_SetTranslation("ResetTrans");
		SKEL PONML 5;
		Goto See;
	}
}

class OrigRevenantRocket : RevenantTracer replaces RevenantTracer
{
	Default
	{
		Speed 16;
	}
	
	States
	{
	Spawn:
		FATB AABB 1 BRIGHT
		{
			A_Tracer();
			if(GetCVar("sgd_spawntrails") == 1) { A_SpawnItem("SGDProjectileTrail",-20,0,-vel.z+10); }
		}
		Loop;
	}
}

class SGDRevenantRocket : RevenantTracer
{
	Default
	{
		Speed 20;
		-SEEKERMISSILE;
	}

	States
	{
	Spawn:
		FATB AABB 1 BRIGHT
		{
			A_Weave(3,3,1.065,1.065);
			if(CVar.FindCVar('sgd_spawntrails').GetBool() == 1) { A_SpawnItem("SGDProjectileTrail",-20,0,-vel.z+10); }
		}
		Loop;
	}
}
