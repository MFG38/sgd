class SGDImp : SGDMonster replaces DoomImp
{
	Default
	{
		Health 60;
		Radius 20;
		Height 56;
		Mass 100;
		Speed 8;
		PainChance 200;
		SeeSound "imp/sight";
		PainSound "imp/pain";
		DeathSound "imp/death";
		ActiveSound "imp/active";
		HitObituary "$OB_IMPHIT";
		Obituary "$OB_IMP";
	}
	States
	{
	Spawn:
		//TROO A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		TROO AB 10 A_Look;
		Loop;
	AltSpawn:
		TROO AABBCCDD 3 A_Wander;
		TROO A 0 A_Look;
		Loop;
	See:
		TROO AABBCCDD 3 A_Chase;
		Loop;
	Melee:
	Missile:
		TROO EF 8 A_FaceTarget;
		TROO G 6 A_TroopAttack ;
		Goto See;
	Pain:
		TROO H 2;
		TROO H 2 A_Pain;
		Goto See;
	Pain.HolyWater:
		TROO H 2 A_SetTranslation("HolyWaterTrans");
		TROO H 2 A_Pain;
		Goto See;
	Pain.Fire:
		TROO H 2 A_SetTranslation("CharredTrans");
		TROO H 2 A_Pain;
		Goto See;
	Death:
		TROO I 8
		{
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
		}
		TROO J 8 A_Scream;
		TROO K 6;
		TROO L 6 A_NoBlocking;
		TROO M -1;
		Stop;
	Death.HolyWater:
		TROO I 8
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
		}
		TROO J 8 A_Scream;
		TROO K 6;
		TROO L 6 A_NoBlocking;
		TROO M -1;
		Stop;
	Death.Fire:
		TROO I 8
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
		}
		TROO J 8 A_Scream;
		TROO K 6;
		TROO L 6 A_NoBlocking;
		TROO M -1;
		Stop;
	XDeath:
		TROO N 5
		{
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
		}
		TROO O 5 A_XScream;
		TROO P 5;
		TROO Q 5 A_NoBlocking;
		TROO RST 5;
		TROO U -1;
		Stop;
	XDeath.HolyWater:
		TROO N 5
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
		}
		TROO O 5 A_XScream;
		TROO P 5;
		TROO Q 5 A_NoBlocking;
		TROO RST 5;
		TROO U -1;
		Stop;
	XDeath.Fire:
		TROO N 5
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
		}
		TROO O 5 A_XScream;
		TROO P 5;
		TROO Q 5 A_NoBlocking;
		TROO RST 5;
		TROO U -1;
		Stop;
	Death.Sploder:
	XDeath.Sploder:
		TROO H 2 A_Pain;
		TROO HHHHHHHHHH 5;
		TROO N 5
		{
			A_GiveToTarget("ImpsKilled");
			A_GiveToTarget("FinisherToken",12);
			A_SpawnItemEx("Explosion",0,0,35);
			//if(FindInventory("ChainExploder")) { A_RadiusGive("ChainExploder",192,RGF_MONSTERS); }
		}
		TROO O 5
		{
			A_XScream();
			A_Explode(666,192,XF_NOTMISSILE);
		}
		TROO P 5;
		TROO Q 5 A_NoBlocking;
		TROO RST 5;
		TROO U -1;
		Stop;
	Raise:
		TROO ML 8 A_SetTranslation("ResetTrans");
		TROO KJI 6;
		Goto See;
	}
}

class SGDImpBall : DoomImpBall replaces DoomImpBall
{
	Default
	{
		Speed 15;
	}
	
	States
	{
	Spawn:
		BAL1 AAAABBBB 1 BRIGHT { if(CVar.FindCVar('sgd_spawntrails').GetBool() == 1) { A_SpawnItemEx("SGDProjectileTrail",-20,0,-vel.z); }}
		Loop;
	}
}
