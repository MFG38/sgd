class SGDFlameBoi : SGDMonster replaces Archvile
{
	Default
	{
		Health 700;
		Radius 20;
		Height 56;
		Mass 500;
		Speed 15;
		PainChance 10;
		MaxTargetRange 896;
		+QUICKTORETALIATE;
		+NOTARGET;
		SeeSound "vile/sight";
		PainSound "vile/pain";
		DeathSound "vile/death";
		ActiveSound "vile/active";
		MeleeSound "vile/stop";
		Obituary "$OB_VILE";
	}
	States
	{
	Spawn:
		VILE A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		VILE AB 10 A_Look;
		Loop;
	AltSpawn:
		VILE AABBCCDDEEFF 2 A_Wander;
		VILE A 0 A_Look;
		Loop;
	See:
		VILE AABBCCDDEEFF 2 A_VileChase;
		Loop;
	Missile:
		VILE G 0 BRIGHT A_VileStart;
		VILE G 10 BRIGHT A_FaceTarget;
		VILE H 8 BRIGHT A_VileTarget;
		VILE IJKLMN 8 BRIGHT A_FaceTarget;
		VILE O 8 BRIGHT A_VileAttack;
		VILE P 20 BRIGHT;
		Goto See;
	Heal:
		VILE [\] 10 BRIGHT;
		Goto See;
	Pain:
		VILE Q 5;
		VILE Q 5 A_Pain;
		Goto See;
	Pain.HolyWater:
		VILE Q 5 A_SetTranslation("HolyWaterTrans");
		VILE Q 5 A_Pain;
		Goto See;
	Pain.Fire:
		VILE Q 5 A_SetTranslation("CharredTrans");
		VILE Q 5 A_Pain;
		Goto See;
	Death:
		VILE Q 7
		{
			A_GiveToTarget("ArchvilesKilled");
			A_GiveToTarget("FinisherToken",65);
		}
		VILE R 7 A_Scream;
		VILE S 7 A_NoBlocking;
		VILE TUVWXY 7;
		VILE Z -1;
		Stop;
	Death.HolyWater:
		VILE Q 7
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("ArchvilesKilled");
			A_GiveToTarget("FinisherToken",65);
		}
		VILE R 7 A_Scream;
		VILE S 7 A_NoBlocking;
		VILE TUVWXY 7;
		VILE Z -1;
		Stop;
	Death.Fire:
		VILE Q 7
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("ArchvilesKilled");
			A_GiveToTarget("FinisherToken",65);
		}
		VILE R 7 A_Scream;
		VILE S 7 A_NoBlocking;
		VILE TUVWXY 7;
		VILE Z -1;
		Stop;
	}
}
