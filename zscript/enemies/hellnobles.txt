class SGDHellKnight : SGDMonster replaces HellKnight
{
	Default
	{
		Health 500;
		Radius 24;
		Height 64;
		Mass 1000;
		Speed 8;
		PainChance 50;
		SeeSound "knight/sight";
		ActiveSound "knight/active";
		PainSound "knight/pain";
		DeathSound "knight/death";
		HitObituary "$OB_KNIGHTHIT";
		Obituary "$OB_KNIGHT";
		BloodColor "00 40 00";
	}
	States
	{
	Spawn:
		//BOS2 A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		BOS2 AB 10 A_Look;
		Loop;
	AltSpawn:
		BOS2 AABBCCDD 3 A_Wander;
		BOS2 A 0 A_Look;
		Loop;
	See:
		BOS2 AABBCCDD 3 A_Chase;
		Loop;
	Melee:
	Missile:
		BOS2 EF 8 A_FaceTarget;
		BOS2 G 8 A_BruisAttack;
		Goto See;
	Pain:
		BOS2 H  2;
		BOS2 H  2 A_Pain;
		Goto See;
	Pain.HolyWater:
		BOS2 H  2 A_SetTranslation("HolyWaterTrans");
		BOS2 H  2 A_Pain;
		Goto See;
	Pain.Fire:
		BOS2 H  2 A_SetTranslation("CharredTrans");
		BOS2 H  2 A_Pain;
		Goto See;
	Death:
		BOS2 I  8
		{
			A_GiveToTarget("HellKnightsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		BOS2 J  8 A_Scream;
		BOS2 K  8;
		BOS2 L  8 A_NoBlocking;
		BOS2 MN 8;
		BOS2 O -1;
		Stop;
	Death.HolyWater:
		BOS2 I  8
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("HellKnightsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		BOS2 J  8 A_Scream;
		BOS2 K  8;
		BOS2 L  8 A_NoBlocking;
		BOS2 MN 8;
		BOS2 O -1;
		Stop;
	Death.Fire:
		BOS2 I  8
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("HellKnightsKilled");
			A_GiveToTarget("FinisherToken",25);
		}
		BOS2 J  8 A_Scream;
		BOS2 K  8;
		BOS2 L  8 A_NoBlocking;
		BOS2 MN 8;
		BOS2 O -1;
		Stop;
	Death.Sploder:
		BOS2 H  2 A_Pain;
		BOS2 HHHHHHHHHH 5;
		BOS2 I  8
		{
			A_GiveToTarget("HellKnightsKilled");
			A_GiveToTarget("FinisherToken",25);
			A_SpawnItemEx("Explosion",0,0,35);
		}
		BOS2 J  8
		{
			A_Scream();
			A_Explode(666,192,XF_NOTMISSILE);
		}
		BOS2 K  8;
		BOS2 L  8 A_NoBlocking;
		BOS2 MN 8;
		BOS2 O -1;
		Stop;
	Raise:
		BOS2 O 8 A_SetTranslation("ResetTrans");
		BOS2 NMLKJI  8;
		Goto See;
	}
}
