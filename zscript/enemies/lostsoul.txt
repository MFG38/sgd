class SGDLostSoul : SGDMonster replaces LostSoul
{
	Default
	{
		Health 100;
		Radius 16;
		Height 56;
		Mass 50;
		Speed 8;
		Damage 3;
		PainChance 256;
		+FLOAT;
		+NOGRAVITY;
		+MISSILEEVENMORE;
		+DONTFALL;
		+ZDOOMTRANS;
		-FLOORCLIP;
		+NOBLOOD;
		AttackSound "skull/melee";
		PainSound "skull/pain";
		DeathSound "skull/death";
		ActiveSound "skull/active";
		RenderStyle "SoulTrans";
		Obituary "$OB_SKULL";
		DamageFactor "Fire", 0.85;
	}
	States
	{
	Spawn:
		SKUL A 0 A_JumpIf(GetCVar("sgd_spawnbehavior") == 1,"AltSpawn");
		SKUL AB 10 BRIGHT A_Look;
		Loop;
	AltSpawn:
		SKUL AB 6 BRIGHT A_Wander;
		SKUL A 0 A_Look;
		Loop;
	See:
		SKUL AB 6 BRIGHT A_Chase;
		Loop;
	Missile:
		SKUL C 6 BRIGHT A_FaceTarget;
		SKUL D 4 BRIGHT A_SkullAttack;
		SKUL CD 4 BRIGHT;
		Goto Missile+2;
	Pain:
		SKUL E 3 BRIGHT;
		SKUL E 3 BRIGHT A_Pain;
		Goto See;
	Pain.HolyWater:
		SKUL E 3 BRIGHT A_SetTranslation("HolyWaterTrans");
		SKUL E 3 BRIGHT A_Pain;
		Goto See;
	Pain.Fire:
		SKUL E 3 BRIGHT A_SetTranslation("CharredTrans");
		SKUL E 3 BRIGHT A_Pain;
		Goto See;
	Death:
		SKUL F 6 BRIGHT
		{
			A_GiveToTarget("LostSoulsKilled");
			A_GiveToTarget("FinisherToken",8);
		}
		SKUL G 6 BRIGHT A_Scream;
		SKUL H 6 BRIGHT;
		SKUL I 6 BRIGHT A_NoBlocking;
		SKUL J 6;
		SKUL K 6;
		Stop;
	Death.HolyWater:
		SKUL F 6 BRIGHT
		{
			A_SetTranslation("HolyWaterTrans");
			A_GiveToTarget("LostSoulsKilled");
			A_GiveToTarget("FinisherToken",8);
		}
		SKUL G 6 BRIGHT A_Scream;
		SKUL H 6 BRIGHT;
		SKUL I 6 BRIGHT A_NoBlocking;
		SKUL J 6;
		SKUL K 6;
		Stop;
	Death.Fire:
		SKUL F 6 BRIGHT
		{
			A_SetTranslation("CharredTrans");
			A_GiveToTarget("LostSoulsKilled");
			A_GiveToTarget("FinisherToken",8);
		}
		SKUL G 6 BRIGHT A_Scream;
		SKUL H 6 BRIGHT;
		SKUL I 6 BRIGHT A_NoBlocking;
		SKUL J 6;
		SKUL K 6;
		Stop;
	}
}
